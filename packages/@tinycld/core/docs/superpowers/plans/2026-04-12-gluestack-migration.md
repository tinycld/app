# HeroUI → GlueStack UI v5 Migration Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace HeroUI Native with GlueStack UI v5 alpha (Uniwind path) for better web compatibility and a broader component set.

**Architecture:** Copy-paste GlueStack components into `ui/` via CLI, rebuild `useThemeColor` on Uniwind's `useCSSVariable`, migrate ~90 files from HeroUI imports to local GlueStack equivalents. Theme tokens stay as CSS variables in `global.css` with Tailwind v4 syntax.

**Tech Stack:** GlueStack UI v5 alpha, Uniwind, Tailwind CSS v4, Expo 55, React Native 0.83

**Spec:** `docs/superpowers/specs/2026-04-12-gluestack-migration-design.md`

The correct metro.config.cjs should look like:

```js
const { getDefaultConfig } = require('expo/metro-config')

const config = getDefaultConfig(__dirname)

config.resolver.resolveRequest = (context, moduleName, platform) => {

    return context.resolveRequest(context, moduleName, platform)

module.exports = withUniwindConfig(config, {
    cssEntryFile: './global.css',
```

- [ ] **Step 1: Understand the Button API change**

HeroUI Button accepts string children:
```tsx
<Button onPress={fn} size="sm">Label</Button>
```

GlueStack Button requires ButtonText:
```tsx
<Button onPress={fn} size="sm">
    <ButtonText>Label</ButtonText>
</Button>
```

GlueStack Button also uses `action` prop instead of className for semantic colors:
- `action="negative"` instead of `className="bg-danger text-danger-foreground"`
- `action="primary"` for default/accent
- `variant="outline"` | `"solid"` | `"link"`

- [ ] **Step 2: Update imports in each file**

Replace:
```ts
import { Button } from 'heroui-native'
// or within a destructured import
import { Button, useThemeColor } from 'heroui-native'
```

With:
```ts
import { Button, ButtonText } from '~/ui/button'
```

Remove `Button` from any remaining heroui-native import. If heroui-native import becomes empty, delete the line.

- [ ] **Step 3: Update Button JSX in each file**

For each `<Button>` usage, wrap string children in `<ButtonText>`:

```tsx
// Before

>
    {confirmLabel}
</Button>

// After
<Button
    onPress={handleConfirm}

    action="negative"
>

</Button>
```

Apply the same pattern to all files. For each Button:

2. Replace `className="bg-danger..."` with `action="negative"`
3. Replace `className="bg-accent..."` with `action="primary"` (or just remove — primary is default)
4. Keep `size`, `isDisabled`, `onPress` props as-is (same API)

```tsx

    <Dialog.Portal>
        <Dialog.Overlay />
        <Dialog.Content className="w-[320px] p-4">
            {content}
        </Dialog.Content>
    </Dialog.Portal>
</Dialog>

// After (GlueStack Modal)
<Modal isOpen={open} onClose={() => setOpen(false)}>
    <ModalBackdrop />
    <ModalContent className="w-[320px] p-4">

    </ModalContent>
</Modal>
```

- `onOpenChange` → `onClose` (only close callback, not toggle)

```tsx
} from '~/ui/alert-dialog'
import { Button, ButtonText } from '~/ui/button'

interface SuretyGuardProps {
    children: (onOpen: () => void) => ReactNode
    message?: string
    confirmLabel?: string
    onConfirmed: () => void | Promise<void>
}

export function SuretyGuard({
    children,
    message = 'Are you sure? This cannot be undone.',
    confirmLabel = 'Yes',
    onConfirmed,
}: SuretyGuardProps) {
    const [open, setOpen] = useState(false)
    const [pending, setPending] = useState(false)

    const handleConfirm = async () => {
        setPending(true)
        try {
            await onConfirmed()
        } finally {
            setPending(false)
            setOpen(false)
        }
    }

    return (
        <>
            {children(() => setOpen(true))}
            <AlertDialog isOpen={open} onClose={() => setOpen(false)}>
                <AlertDialogBackdrop />
                <AlertDialogContent className="w-[320px]">
                    <AlertDialogBody className="pt-4">
                        <Text className="text-sm text-foreground">{message}</Text>
                    </AlertDialogBody>
                    <AlertDialogFooter>
                        <View className="flex-row gap-3 justify-end">
                            <Pressable
                                disabled={pending}
                                onPress={() => setOpen(false)}
                                style={{ padding: 8 }}
                            >
                                <Text className="text-sm text-foreground">Cancel</Text>
                            </Pressable>
                            <Button
                                onPress={handleConfirm}
                                isDisabled={pending}
                                size="sm"
                                action="negative"
                            >
                                <ButtonText>{confirmLabel}</ButtonText>
                            </Button>
                        </View>
                    </AlertDialogFooter>
                </AlertDialogContent>
            </AlertDialog>
        </>
    )
}

interface ConfirmTrashProps {
    children: (onOpen: () => void) => ReactNode
    itemName: string
    onConfirmed: () => void | Promise<void>
}

export function ConfirmTrash({ children, itemName, onConfirmed }: ConfirmTrashProps) {
    return (
        <SuretyGuard
            message={`Are you sure you want to move "${itemName}" to trash? It will be permanently removed after 30 days.`}
            confirmLabel="Move to trash"
            onConfirmed={onConfirmed}
        >
            {children}
        </SuretyGuard>
    )
}
```
